015class Tictactoe{
016    /**
017     * Main method for the  game.
018     * 
019     * @param args Command line parameters. Not used.
020    */
021    public static void main(String[] args) {
022
023        //Creates object for game
024        Playground game = new Playground();
025
026        /**
027         * boolean to used to define have
028         * player or ai won
029         */
030        boolean on = true;
031
032        /**
033         * Playground method checkwin to use
034         * boolean on when player or ai wins
035         * while it's true game is on.
036         * Playground methods Playermove,
037         * AImove and printPG used to play the
038         * game
039         */
040        while(game.checkwin(on)){
041            
042            game.playerMove();
043            game.AImove();
044            game.printPG();
045
046        }
047
048        //who won game
049        System.out.println(game.winner());
050
051    }
